CREATE TABLE Branch(                                -- Create Branch table
branch_name VARCHAR(50) PRIMARY KEY,               -- Branch name as primary key
branch_city VARCHAR(50) NOT NULL,                  -- Store branch city
asset DECIMAL(10,2)                                -- Store branch asset
);

CREATE TABLE Account(                               -- Create Account table
Acc_no INT PRIMARY KEY,                             -- Account number as primary key
branch_name VARCHAR(50),                            -- Store branch name
balance DECIMAL(10,2) NOT NULL,                     -- Store account balance
FOREIGN KEY(branch_name) REFERENCES Branch(branch_name) -- Foreign key reference
);

CREATE TABLE Customer(                              -- Create Customer table
cust_name VARCHAR(50) PRIMARY KEY,                  -- Customer name as primary key
cust_street VARCHAR(50),                            -- Store customer street
cust_city VARCHAR(50)                               -- Store customer city
);

CREATE TABLE Depositor(                             -- Create Depositor table
cust_name VARCHAR(50),                              -- Store customer name
Acc_no INT,                                         -- Store account number
PRIMARY KEY(cust_name, Acc_no),                     -- Composite primary key
FOREIGN KEY(cust_name) REFERENCES Customer(cust_name), -- Foreign key for customer
FOREIGN KEY(Acc_no) REFERENCES Account(Acc_no)      -- Foreign key for account
);

CREATE TABLE Loan(                                  -- Create Loan table
loan_no INT PRIMARY KEY,                            -- Loan number as primary key
branch_name VARCHAR(50),                            -- Store branch name
amount DECIMAL(10,2) NOT NULL,                      -- Store loan amount
FOREIGN KEY(branch_name) REFERENCES Branch(branch_name) -- Foreign key reference
);

CREATE TABLE Borrower(                              -- Create Borrower table
cust_name VARCHAR(50),                              -- Store customer name
loan_no INT,                                        -- Store loan number
PRIMARY KEY(cust_name, loan_no),                    -- Composite primary key
FOREIGN KEY(cust_name) REFERENCES Customer(cust_name), -- Foreign key for customer
FOREIGN KEY(loan_no) REFERENCES Loan(loan_no)       -- Foreign key for loan
);
