// Calculate total salary of employees for each company
var mapFunction = function() {
emit(this.Company_name, this.Salary);
};

var reduceFunction = function(key, values) {
return Array.sum(values);
};

db.Employee.mapReduce(
mapFunction,
reduceFunction,
{
out:"TotalSalaryPerCompany"
}
)

db.TotalSalaryPerCompany.find()

// Calculate average salary of employees whose city is Pune
var mapFunction = function() {
if(this.Address.City=="Pune")
emit(this.Company_name, {total:this.Salary, count:1});
};

var reduceFunction = function(key, values) {
var result = {total:0, count:0};

values.forEach(function(v){
result.total += v.total;
result.count += v.count;
});

return result;
};

var finalizeFunction = function(key, reducedValue) {
return reducedValue.total / reducedValue.count;
};

db.Employee.mapReduce(
mapFunction,
reduceFunction,
{
out:"AverageSalaryPune",
finalize:finalizeFunction
}
)

db.AverageSalaryPune.find()
