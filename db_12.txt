CREATE PROCEDURE library_fine()

BEGIN

    DECLARE v_rollno INT;                   -- variable for roll number
    DECLARE v_bookname VARCHAR(50);         -- variable for book name
    DECLARE v_days INT;                     -- variable to store number of days
    DECLARE v_fine DOUBLE DEFAULT 0;        -- variable to store fine amount
    DECLARE v_issue_date DATE;              -- variable for issue date

    SET v_rollno = 101;                     -- enter roll number
    SET v_bookname = 'DBMS';                -- enter book name

    SELECT DateofIssue
    INTO v_issue_date
    FROM Borrower
    WHERE Rollin = v_rollno
    AND NameofBook = v_bookname;

    SET v_days = DATEDIFF(CURDATE(), v_issue_date);
                                            -- calculate days from issue date

    IF v_days BETWEEN 15 AND 30 THEN

        SET v_fine = v_days * 5;            -- Rs 5 per day fine

    ELSEIF v_days > 30 THEN

        SET v_fine = (30 * 5) + ((v_days - 30) * 50);
                                            -- first 30 days Rs 5/day
                                            -- remaining days Rs 50/day

    END IF;

    UPDATE Borrower
    SET Status = 'R'                        -- change status from I to R
    WHERE Rollin = v_rollno
    AND NameofBook = v_bookname;

    IF v_fine > 0 THEN

        INSERT INTO Fine(Roll_no, Date, Amt)
        VALUES(v_rollno, CURDATE(), v_fine);
                                            -- insert fine details into fine table

    END IF;

    SELECT v_fine AS Fine_Amount;           -- display fine amount

END$$

DELIMITER ;

CALL library_fine();

DECLARE

    r_no NUMBER;
    b_name VARCHAR2(50);

    issue_date DATE;
    days NUMBER;
    fine_amt NUMBER := 0;

BEGIN

    r_no := &Roll_No;
    b_name := '&Book_Name';

    SELECT DateofIssue
    INTO issue_date
    FROM Borrower
    WHERE Roll_no = r_no
    AND NameofBook = b_name;

    days := SYSDATE - issue_date;

    IF days BETWEEN 15 AND 30 THEN

        fine_amt := days * 5;

    ELSIF days > 30 THEN

        fine_amt := (30 * 5) + ((days - 30) * 50);

    END IF;

    UPDATE Borrower
    SET Status = 'R'
    WHERE Roll_no = r_no
    AND NameofBook = b_name;

    IF fine_amt > 0 THEN

        INSERT INTO Fine
        VALUES(r_no, SYSDATE, fine_amt);

    END IF;

    COMMIT;

    DBMS_OUTPUT.PUT_LINE('Fine Amount = ' || fine_amt);

END;
