CREATE PROCEDURE Check_Attendance(IN v_roll INT)

BEGIN

    DECLARE v_attendance INT;

    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        SELECT 'Error Occurred' AS MESSAGE;
    END;

    SELECT Attendance
    INTO v_attendance
    FROM Student
    WHERE Roll = v_roll;

    IF v_attendance < 75 THEN

        UPDATE Student
        SET Status = 'Detained'
        WHERE Roll = v_roll;

        SELECT 'Term not granted' AS MESSAGE;

    ELSE

        UPDATE Student
        SET Status = 'Not Detained'
        WHERE Roll = v_roll;

        SELECT 'Term granted' AS MESSAGE;

    END IF;

END$$

DELIMITER ;
CALL Check_Attendance(1);

DECLARE

    r_no NUMBER;
    att NUMBER;

BEGIN

    r_no := &Roll_No;

    SELECT Attendance
    INTO att
    FROM Student
    WHERE Roll = r_no;

    IF att < 75 THEN

        UPDATE Student
        SET Status = 'Detained'
        WHERE Roll = r_no;

        DBMS_OUTPUT.PUT_LINE('Term Not Granted');

    ELSE

        UPDATE Student
        SET Status = 'Not Detained'
        WHERE Roll = r_no;

        DBMS_OUTPUT.PUT_LINE('Term Granted');

    END IF;

    COMMIT;

EXCEPTION

    WHEN NO_DATA_FOUND THEN

        DBMS_OUTPUT.PUT_LINE('Student Record Not Found');

    WHEN OTHERS THEN

        DBMS_OUTPUT.PUT_LINE('Error Occurred');

END;
