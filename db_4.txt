CREATE TABLE Cust_Master(                           -- Create customer master table
Cust_no VARCHAR(10) PRIMARY KEY,                    -- Customer number as primary key
Cust_name VARCHAR(50),                              -- Store customer name
Cust_addr VARCHAR(100)                              -- Store customer address
);

CREATE TABLE Orders(                                -- Create Orders table
Order_no INT PRIMARY KEY,                           -- Order number as primary key
Cust_no VARCHAR(10),                                -- Store customer number
Order_date DATE,                                    -- Store order date
Qty_Ordered INT,                                    -- Store ordered quantity
FOREIGN KEY(Cust_no) REFERENCES Cust_Master(Cust_no) -- Foreign key reference
);

CREATE TABLE Product(                               -- Create Product table
Product_no INT PRIMARY KEY,                         -- Product number as primary key
Product_name VARCHAR(50),                           -- Store product name
Order_no INT,                                       -- Store order number
FOREIGN KEY(Order_no) REFERENCES Orders(Order_no)   -- Foreign key reference
);

SELECT Cust_Master.Cust_name, Product.Product_name  -- Display customer name and purchased product
FROM Cust_Master
JOIN Orders
ON Cust_Master.Cust_no = Orders.Cust_no
JOIN Product
ON Orders.Order_no = Product.Order_no;


CREATE VIEW View1 AS                                -- Create View1
SELECT Cust_Master.Cust_name, Product.Product_name
FROM Cust_Master
JOIN Orders
ON Cust_Master.Cust_no = Orders.Cust_no
JOIN Product
ON Orders.Order_no = Product.Order_no;


SELECT Cust_Master.Cust_name, Product.Product_name, Orders.Qty_Ordered -- Display product and quantity
FROM Cust_Master
JOIN Orders
ON Cust_Master.Cust_no = Orders.Cust_no
JOIN Product
ON Orders.Order_no = Product.Order_no;


SELECT *                                            -- Perform inner join operation
FROM Cust_Master
INNER JOIN Orders
ON Cust_Master.Cust_no = Orders.Cust_no;


SELECT *                                            -- Perform left join operation
FROM Cust_Master
LEFT JOIN Orders
ON Cust_Master.Cust_no = Orders.Cust_no;


SELECT *                                            -- Perform right join operation
FROM Cust_Master
RIGHT JOIN Orders
ON Cust_Master.Cust_no = Orders.Cust_no;
