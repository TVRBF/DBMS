CREATE TABLE Employee(                              -- Create Employee table
emp_id INT PRIMARY KEY,                             -- Employee ID as primary key
employee_name VARCHAR(50),                          -- Store employee name
street VARCHAR(50),                                 -- Store street name
city VARCHAR(50)                                    -- Store city name
);

CREATE TABLE Company(                               -- Create Company table
company_name VARCHAR(50) PRIMARY KEY,               -- Company name as primary key
city VARCHAR(50)                                    -- Store company city
);

CREATE TABLE Works(                                 -- Create Works table
employee_name VARCHAR(50),                          -- Store employee name
company_name VARCHAR(50),                           -- Store company name
salary DECIMAL(10,2),                               -- Store employee salary
FOREIGN KEY(company_name) REFERENCES Company(company_name) -- Foreign key reference
);

CREATE TABLE Manages(                               -- Create Manages table
employee_name VARCHAR(50),                          -- Store employee name
manager_name VARCHAR(50)                            -- Store manager name
);

FROM Works
ORDER BY company_name ASC, employee_name DESC;
