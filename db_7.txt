CREATE TABLE CUSTOMERS(                             -- Create CUSTOMERS table
CNo INT PRIMARY KEY,                                -- Customer number as primary key
Cname VARCHAR(50),                                  -- Store customer name
Ccity VARCHAR(50),                                  -- Store customer city
CMobile VARCHAR(15)                                 -- Store customer mobile number
);

CREATE TABLE ITEMS(                                 -- Create ITEMS table
INo INT PRIMARY KEY,                                -- Item number as primary key
Iname VARCHAR(50),                                  -- Store item name
Itype VARCHAR(50),                                  -- Store item type
Iprice DECIMAL(10,2),                               -- Store item price
Icount INT                                          -- Store item count
);

CREATE TABLE PURCHASE(                              -- Create PURCHASE table
PNo INT PRIMARY KEY,                                -- Purchase number as primary key
Pdate DATE,                                         -- Store purchase date
Pquantity INT,                                      -- Store purchase quantity
CNo INT,                                            -- Store customer number
INo INT,                                            -- Store item number
FOREIGN KEY(CNo) REFERENCES CUSTOMERS(CNo),         -- Foreign key for customer
FOREIGN KEY(INo) REFERENCES ITEMS(INo)              -- Foreign key for item
);

SELECT *                                            -- Display stationary items within price range
FROM ITEMS
WHERE Itype='Stationary'
AND Iprice BETWEEN 400 AND 1000;


UPDATE CUSTOMERS                                    -- Update mobile number
SET CMobile='9999999999'
WHERE Cname='Gopal';


SELECT *                                            -- Display item with maximum price
FROM ITEMS
WHERE Iprice=
(
SELECT MAX(Iprice)
FROM ITEMS
);


SELECT *                                            -- Display purchases from recent to oldest
FROM PURCHASE
ORDER BY Pdate DESC;


SELECT Ccity, COUNT(*) AS Total_Customers           -- Count customers city-wise
FROM CUSTOMERS
GROUP BY Ccity;


SELECT PURCHASE.Pquantity                           -- Display purchased quantity of Maya
FROM PURCHASE
JOIN CUSTOMERS
ON PURCHASE.CNo = CUSTOMERS.CNo
WHERE CUSTOMERS.Cname='Maya';


CREATE VIEW Stationary_View AS                      -- Create view for stationary items
SELECT Iname, Iprice, Icount
FROM ITEMS
WHERE Itype='Stationary';
